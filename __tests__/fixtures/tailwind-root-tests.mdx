## Updating Styling Dynamically

import { useRef, useEffect } from 'react';

```mdx
export const Dynamic = () => {
  const ref = useRef();

useEffect(() => {
ref.current.classList.add('bg-gray-950', 'text-green-500', 'p-5', 'rounded-xl');
}, [])

return <div ref={ref}>Dynamic Component</div>
};
```
